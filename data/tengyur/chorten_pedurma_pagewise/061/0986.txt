མཐོང་ནས་ཁྲོ་བ་བཟློག    །ལུས་ཅན་རྣམས་ལ་སེམས་ཡོད་དང་།      །སེམས་མེད་གཉི་གས་གནོད་བྱས་ན།      །སེམས་ཡོད་ཅི་སྟེ་བཀར་ཏེ་བཀོན།    །དེ་བས་གནོད་པ་བཟོད་པར་གྱིས།      །ལ་ལ་རྨོངས་པས་ཉེས་པ་བྱེད།      །ལ་ལ་རྨོངས་ཏེ་ཁྲོས་གྱུར་ན།    །དེ་ལ་སྐྱོན་མེད་གང་གིས་བྱ།    །སྐྱོན་དང་བཅས་ཏེ་གང་ཞིག་ཡིན།    །གང་གིས་གཞན་དག་གནོད་བྱེད་པའི།    །ལས་དེ་སྔོན་ཆད་ཅི་ཕྱིར་བྱས།    །ཐམས་ཅད་ལས་ལ་རག་ལས་ན།    །བདག་གིས་འདི་ལ་ཅི་སྟེ་བཀོན།།དེ་ལྟར་མཐོང་ནས་ཅི་ནས་ཀྱང་།    །ཐམས་ཅད་ཕན་ཚུན་བྱམས་སེམས་སུ།    །འགྱུར་བ་དེ་ལྟར་བདག་གིས་ནི།    །བསོད་ནམས་དག་ལ་བསྒྲིམས་ཏེ་བྱ།    །དཔེར་ན་ཁྱིམ་ཚིག་གྱུར་པའི་མེ།      །ཁྱིམ་གཞན་ཞིག་ཏུ་སོང་ནས་ནི།      །རྩྭ་སོགས་གང་ལ་མཆེད་བྱེད་པ།    །དེ་ནི་ཕྱུང་སྟེ་དོར་བ་ཡིན།    །དེ་བཞིན་གང་ལ་སེམས་ཆགས་ན།    །ཞེ་སྡང་མེ་ནི་མཆེད་གྱུར་པ།    །བསོད་ནམས་ཚིག་པར་དོགས་པ་ཡིས།    །དེ་ནི་དེ་ཡི་མོད་ལ་དོར།    །་གསད་བྱའི་མི་ཞིག་ལག་བཅད་དེ༑    །གལ་ཏེ་ཐར་ན་ཅིས་མ་ལེགས།    །གལ་ཏེ་མི་ཡི་སྡུག་བསྔལ་གྱིས།    །དམྱལ་བ་བྲལ་ན་ཅིས་མ་ལེགས།    །ད་ལྟའི་སྡུག་བསྔལ་འདི་ཙམ་ལའང་།    །བདག་གིས་བཟོད་པར་མི་ནུས་ན།    །དེས་ན་དམྱལ་བའི་སྡུག་བསྔལ་རྒྱུ།    །ཁྲོ་བ་ཅི་སྟེ་བཟློག་མི་བྱེད།    །འདོད་པའི་དོན་དུ་སྲེག་ལ་སོགས།    །དམྱལ་བར་སྟོང་ཕྲག་མྱོང་གྱུར་ཡང་།    །བདག་གིས་རང་གི་དོན་དང་ནི།    །གཞན་གྱི་དོན་ཡང་མ་བྱས་སོ།    །འདི་ནི་དེ་ཙམ་གནོད་མིན་ལ།    །དོན་ཆེན་དག་ཀྱང་འགྲུབ་འགྱུར་བས།    །འགྲོ་བའི་གནོད་སེལ་སྡུག་བསྔལ་ལ།    །དགའ་བ་འབའ་ཞིག་འདིར་བྱ་རིགས།    །གཞན་གྱིས་ཡོན་ཏན་ལྡན་བསྟོད་ནས།    །གལ་ཏེ་དགའ་བའི་བདེ་ཐོབ་ན།    །ཡིད་ཁྱོད་ཀྱང་ནི་དེ་བསྟོད་ནས།    །ཅི་ཕྱིར་དེ་ལྟར་དགའ་
[ བསྟན་འགྱུར།    དབུ་མ།  ལ  ༡༢༠   དེབ།61-ལྡེབ།988 ]
